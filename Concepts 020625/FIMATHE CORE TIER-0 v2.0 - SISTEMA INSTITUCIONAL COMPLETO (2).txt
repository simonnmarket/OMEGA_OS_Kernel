@dataclass
class VerificationEvidence:
    """Estrutura padrão para evidências de verificação TIER-0"""
    evidence_type: str
    content: Any
    timestamp: str
    checksum: str
    source: str
    protocol_version: str = "2.0"
    
    def validate_checksum(self) -> bool:
        """Valida integridade da evidência com SHA3-256"""
        content_str = json.dumps(self.content, sort_keys=True, default=str)
        calculated = hashlib.sha3_256(content_str.encode()).hexdigest()
        integrity_check = calculated == self.checksum
        if not integrity_check:
            logger.critical(f"CHECKSUM FAILED: Expected {self.checksum[:16]}..., Got {calculated[:16]}...")
        return integrity_check

class ShieldSystem:
    """
    SISTEMA DE BLINDAGEM TIER-0
    Implementa SHIELD_ON() conforme Protocolo Omega
    """
    
    def __init__(self):
        self.shield_active = False
        self.integrity_checksum = None
        self.last_backup = None
        self.emergency_mode = False
        self.checksum_file = "fimathe_core_checksum.sha3"
        
    def SHIELD_ON(self, system_components: Dict) -> Tuple[bool, List[str]]:
        """
        Ativa blindagem institucional completa
        Retorna (status, lista_de_falhas)
        """
        failures = []
        
        # 1. VERIFICAÇÃO DE INTEGRIDADE
        if not self.verify_integrity():
            failures.append("INTEGRITY_CHECK_FAILED")
            logger.critical("Integrity verification failed - possible code tampering")
        
        # 2. VERIFICAÇÃO DE DEPENDÊNCIAS
        if not self._verify_dependencies():
            failures.append("DEPENDENCY_CHECK_FAILED")
        
        # 3. VERIFICAÇÃO DE MERCADO
        if not self._check_market_conditions():
            failures.append("MARKET_CONDITIONS_INVALID")
        
        # 4. VERIFICAÇÃO DE CAPITAL
        if not self._check_capital_requirements(system_components.get('capital', 0)):
            failures.append("CAPITAL_REQUIREMENTS_NOT_MET")
        
        # 5. VERIFICAÇÃO DE RISCO
        if not self._check_risk_limits(system_components):
            failures.append("RISK_LIMITS_EXCEEDED")
        
        # 6. BACKUP AUTOMÁTICO
        if not self._perform_backup():
            failures.append("BACKUP_FAILED")
        
        # STATUS FINAL
        if not failures:
            self.shield_active = True
            self._generate_activation_log()
            logger.info("SHIELD_ON() ACTIVATED - All systems nominal")
            return True, []
        else:
            self.emergency_mode = True
            self._trigger_fail_safe(failures)
            return False, failures
    
    def verify_integrity(self) -> bool:
        """
        Verifica integridade do código com SHA3-256
        Conforme /rules --apply tier0 --checksum sha3-256
        """
        try:
            # Calcular checksum do script atual
            with open(__file__, 'rb') as f:
                current_checksum = hashlib.sha3_256(f.read()).hexdigest()
            
            self.integrity_checksum = current_checksum
            
            # Verificar contra checksum armazenado (se existir)
            if os.path.exists(self.checksum_file):
                with open(self.checksum_file, 'r') as f:
                    stored_checksum = f.read().strip()
                
                if current_checksum != stored_checksum:
                    logger.warning(f"Checksum mismatch: Current {current_checksum[:16]}..., Stored {stored_checksum[:16]}...")
                    # Em produção, isso seria crítico
                    # Em desenvolvimento, podemos atualizar
                    self._update_checksum(current_checksum)
                    return True  # Permite desenvolvimento
                else:
                    logger.info(f"Integrity verified: SHA3-256 {current_checksum[:16]}... OK")
                    return True
            else:
                # Primeira execução - armazenar checksum
                self._update_checksum(current_checksum)
                logger.info(f"Initial checksum stored: {current_checksum[:16]}...")
                return True
                
        except Exception as e:
            logger.error(f"Integrity check failed: {e}")
            return False
    
    def _update_checksum(self, checksum: str):
        """Atualiza arquivo de checksum"""
        with open(self.checksum_file, 'w') as f:
            f.write(checksum)
    
    def _verify_dependencies(self) -> bool:
        """Verifica dependências do sistema"""
        required_modules = ['MetaTrader5', 'pandas', 'numpy']
        missing = []
        
        for module in required_modules:
            try:
                __import__(module.lower() if module == 'MetaTrader5' else module)
            except ImportError:
                missing.append(module)
        
        if missing:
            logger.error(f"Missing dependencies: {missing}")
            return False
        
        return True
    
    def _check_market_conditions(self) -> bool:
        """Verifica condições de mercado"""
        now = datetime.now()
        
        # Não operar fins de semana (exceto domingo M15 configurado separadamente)
        if now.weekday() >= 5:  # Sábado ou Domingo
            logger.info("Market closed (weekend)")
            return False
        
        # Verificar horário de trading (exemplo: não operar fora do horário principal)
        current_hour = now.hour
        if current_hour < 1 or current_hour > 23:  # Horário reduzido
            logger.warning("Outside main trading hours")
            return False
        
        return True
    
    def _check_capital_requirements(self, capital: float) -> bool:
        """Verifica requisitos mínimos de capital"""
        min_capital = 10000  # Mínimo para operação segura
        
        if capital < min_capital:
            logger.error(f"Insufficient capital: ${capital} < ${min_capital} minimum")
            return False
        
        return True
    
    def _check_risk_limits(self, components: Dict) -> bool:
        """Verifica limites de risco"""
        # Verificar se stops semanais não excedidos
        stops_semana = components.get('stops_semana', 0)
        max_stops = components.get('max_stops_semana', 2)
        
        if stops_semana >= max_stops:
            logger.warning(f"Weekly stops limit reached: {stops_semana}/{max_stops}")
            return False
        
        return True
    
    def _perform_backup(self) -> bool:
        """Executa backup do sistema (simulado)"""
        try:
            self.last_backup = datetime.now(timezone.utc).isoformat()
            # Em produção, isso enviaria para AWS S3 ou similar
            logger.info(f"Backup performed at {self.last_backup}")
            return True
        except Exception as e:
            logger.error(f"Backup failed: {e}")
            return False
    
    def _generate_activation_log(self):
        """Gera log de ativação do shield"""
        activation_log = {
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "shield_status": "ACTIVE",
            "integrity_checksum": self.integrity_checksum[:32] if self.integrity_checksum else None,
            "backup_timestamp": self.last_backup,
            "system_components": ["CanalAbertura", "PrimeiroCiclo", "GestaoFimathe", "MultiAgent"]
        }
        
        log_file = "shield_activation_log.json"
        with open(log_file, 'a') as f:
            json.dump(activation_log, f)
            f.write('\n')
    
    def _trigger_fail_safe(self, failures: List[str]):
        """
        Ativa protocolo de fail-safe
        Conforme Emergency Protocol do Omega
        """
        logger.critical(f"FAIL-SAFE TRIGGERED: {failures}")
        
        # Em produção: fechar todas as posições, notificar admin, etc.

            "shield_status": "FAILED",
            "recommendation": "Manual intervention required"
        }
        
        # Salvar relatório de emergência
        with open("emergency_report.json", 'w') as f:
            json.dump(emergency_report, f, indent=2)

class MultiAgentSystem:

    AGENTS = {

            'status': 'ONLINE',

            'task': 'Otimizar arquitetura Fimathe Core'
        },
        'coder': {
            'role': 'Geração de Código Blindado',
            'status': 'ONLINE',
            'command': '/comet --mode code --api gpt4',
            'task': 'Implementar melhorias e correções'
        },
        'debugger': {
            'role': 'Validação Linha-a-Linha',
            'status': 'ONLINE',
            'command': '/debug --auto-fix --level max',
            'task': 'Auditar lógica e identificar bugs'
        },
        'rule_enforcer': {
            'role': 'Conformidade TIER-0',
            'status': 'ONLINE',
            'command': '/rules --apply tier0 --checksum sha3-256',
            'task': 'Garantir padrões institucionais'
        }
    }
    
    def __init__(self):
        self.agents_active = {}
        self.task_queue = []
        self.boomeran_tasks = []
        
    def initialize_agents(self):
        """Inicializa todos os subagentes"""
        for agent_name, agent_info in self.AGENTS.items():
            self.agents_active[agent_name] = {
                **agent_info,
                'initialized_at': datetime.now(timezone.utc).isoformat(),
                'last_active': datetime.now(timezone.utc).isoformat(),
                'tasks_completed': 0
            }
        
        logger.info(f"MultiAgent System initialized with {len(self.agents_active)} agents")
    
    def assign_task(self, agent: str, task: Dict) -> bool:
        """Atribui tarefa a agente específico"""
        if agent not in self.AGENTS:
            logger.error(f"Unknown agent: {agent}")
            return False
        
        task_with_meta = {
            **task,
            'assigned_to': agent,
            'assigned_at': datetime.now(timezone.utc).isoformat(),
            'status': 'PENDING',
            'task_id': f"TASK-{hashlib.md5(json.dumps(task).encode()).hexdigest()[:8]}"
        }
        
        self.task_queue.append(task_with_meta)
        logger.info(f"Task assigned to {agent}: {task.get('description', 'No description')}")
        
        return True
    
    def create_boomeran_task(self, description: str, subtasks: List[Dict]) -> Dict:
        """
        Cria tarefa Boomeran conforme protocolo
        /boomeran --create --agents 3 --roles coder,reviewer,debugger
        """
        boomeran_task = {
            "boomeran_id": f"BOOM-{datetime.now(timezone.utc).strftime('%Y%m%d%H%M%S')}",
            "created_at": datetime.now(timezone.utc).isoformat(),
            "description": description,
            "subtasks": subtasks,
            "agents_assigned": list(self.AGENTS.keys()),
            "status": "CREATED",
            "progress": 0,
            "checksum": None
        }
        
        # Calcular checksum
        task_str = json.dumps(boomeran_task, sort_keys=True, default=str)
        boomeran_task["checksum"] = hashlib.sha3_256(task_str.encode()).hexdigest()
        
        self.boomeran_tasks.append(boomeran_task)
        logger.info(f"Boomeran task created: {boomeran_task['boomeran_id']}")
        
        return boomeran_task
    
    def get_system_status(self) -> Dict:
        """Retorna status completo do sistema multi-agente"""
        return {
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "total_agents": len(self.agents_active),
            "active_agents": [a for a, info in self.agents_active.items() if info['status'] == 'ONLINE'],
            "pending_tasks": len(self.task_queue),
            "boomeran_tasks": len(self.boomeran_tasks),
            "agents_detail": self.agents_active
        }

def calculate_atr(df: pd.DataFrame, period: int = 14) -> float:
    """
    Calcula o Average True Range (ATR) para SL Dinâmico
    Com fallback seguro para edge cases
    """
    if len(df) < period:
        logger.warning(f"DataFrame com menos de {period} barras. Usando fallback seguro.")
        
        # FALLBACK SEGURO: média do range das últimas velas disponíveis
        if len(df) >= 5:
            recent_range = df['high'].tail(5).values - df['low'].tail(5).values
            fallback_atr = np.mean(recent_range) * 0.5
            logger.info(f"ATR fallback: {fallback_atr:.2f} (baseado em últimas {len(df)} velas)")
            return float(fallback_atr)
        else:
            # Último recurso: valor conservador
            conservative_atr = 50.0  # 50 pontos conservador para XAUUSD
            logger.warning(f"ATR emergency fallback: {conservative_atr:.2f} pontos")
            return conservative_atr
    
    # CÁLCULO ATR PADRÃO
    df_temp = df.copy()
    
    # True Range (TR)
    high_low = df_temp['high'] - df_temp['low']
    high_close_prev = np.abs(df_temp['high'] - df_temp['close'].shift(1))
    low_close_prev = np.abs(df_temp['low'] - df_temp['close'].shift(1))
    
    df_temp['tr'] = high_low.combine(high_close_prev, max).combine(low_close_prev, max)
    
    # Average True Range (EMA)
    atr_series = df_temp['tr'].dropna().ewm(span=period, adjust=False).mean()
    
    atr_value = float(atr_series.iloc[-1]) if not atr_series.empty else 0.0
    
    # Sanity check
    if atr_value <= 0:
        logger.warning(f"ATR calculated as {atr_value}, using fallback")
        return calculate_atr(df, period)  # Recursão com fallback
    
    logger.info(f"ATR{period} calculado: {atr_value:.2f} pontos")
    return atr_value

def verify_mt5_connection(max_retries: int = 3, retry_delay: int = 5) -> bool:
    """
    Verifica e estabelece conexão MT5 com retry automático
    Conforme boas práticas de sistemas críticos
    """
    for attempt in range(max_retries):
        try:
            if mt5.initialize():
                logger.info(f"MT5 connection established (attempt {attempt + 1}/{max_retries})")
                return True
            else:
                error = mt5.last_error()
                logger.warning(f"MT5 initialization failed (attempt {attempt + 1}): {error}")
                
                if attempt < max_retries - 1:
                    logger.info(f"Retrying in {retry_delay} seconds...")
                    time.sleep(retry_delay)
                    
        except Exception as e:
            logger.error(f"MT5 connection exception (attempt {attempt + 1}): {e}")
            if attempt < max_retries - 1:
                time.sleep(retry_delay)
    
    logger.critical(f"Failed to connect to MT5 after {max_retries} attempts")
    return False

class CanalAbertura:
    """
    PILAR FIMATHE: 4 velas abertura → Canal validado
    COM BLINDAGEM: Validação de dados, edge case handling
    """
    
    def __init__(self, min_range_pts: float = 300, max_range_pts: float = 1200):
        self.min_range_pts = min_range_pts    # <300pts = inválido
        self.max_range_pts = max_range_pts    # >1200pts = fatiar
        self.validation_errors = []
        
    def build(self, df: pd.DataFrame, is_domingo_m15: bool = False) -> Dict:
        """
        Constrói canal 4 velas com validação robusta
        """
        self.validation_errors = []
        
        # VALIDAÇÃO DE DADOS DE ENTRADA
        if df.empty or len(df) < 4:
            self.validation_errors.append("INSUFFICIENT_DATA")
            logger.error("DataFrame vazio ou com menos de 4 barras")
            return self._build_error_response("INSUFFICIENT_DATA")
        
        # 4 VELAS DE ABERTURA (robusto a tamanhos variados)
        lookback = min(20, len(df))
        velas_abertura = df.tail(lookback).head(4)
        
        if len(velas_abertura) < 4:
            self.validation_errors.append("INCOMPLETE_OPENING_BARS")
            logger.warning(f"Apenas {len(velas_abertura)} velas de abertura disponíveis")
        
        canal_high = velas_abertura['high'].max()
        canal_low = velas_abertura['low'].min()
        canal_range = canal_high - canal_low
        
        # GAP DETECTION (robusto)
        gap_size = 0.0
        is_gap = False
        if len(df) >= 3:
            try:
                gap_size = abs(df['open'].iloc[-1] - df['close'].iloc[-2])
                is_gap = gap_size > self.min_range_pts * 0.5
            except (IndexError, KeyError) as e:
                logger.warning(f"Gap detection failed: {e}")
        
        # VALIDAÇÃO FIMATHE
        valid_range = self.min_range_pts <= canal_range <= self.max_range_pts
        precisa_fatiar = canal_range > self.max_range_pts
        
        if not valid_range:
            self.validation_errors.append(f"INVALID_RANGE_{canal_range:.0f}")
        
        canal = {
            'high': canal_high,
            'low': canal_low,
            'range_pts': canal_range,
            'range_pct': (canal_range / df['close'].iloc[-1] * 100) if df['close'].iloc[-1] != 0 else 0,
            'gap_size': gap_size,
            'is_gap': is_gap,
            'valid': valid_range,
            'fatiar': precisa_fatiar,
            'is_domingo_m15': is_domingo_m15,
            'velas_count': len(velas_abertura),
            'validation_errors': self.validation_errors,
            'validation_passed': len(self.validation_errors) == 0
        }
        
        status_msg = f"Canal: {canal_range:.1f}pts | Válido: {valid_range}"
        if self.validation_errors:
            logger.warning(f"{status_msg} | Erros: {self.validation_errors}")
        else:
            logger.info(status_msg)
        
        return canal
    
    def _build_error_response(self, error_code: str) -> Dict:
        """Resposta padronizada para erros"""
        return {
            'high': 0,
            'low': 0,
            'range_pts': 0,
            'range_pct': 0,
            'gap_size': 0,
            'is_gap': False,
            'valid': False,
            'fatiar': False,
            'is_domingo_m15': False,
            'velas_count': 0,
            'validation_errors': [error_code],
            'validation_passed': False
        }

class PrimeiroCiclo:
    """
    1º CICLO: Rompimento canal → Direção + Confiança
    COM BLINDAGEM: Threshold dinâmico, confirmação robusta
    """
    
    def __init__(self, rompimento_min_pct: float = 0.3):
        self.rompimento_min_pct = rompimento_min_pct  # 30% mínimo
    
    def detect(self, canal: Dict, df: pd.DataFrame) -> Dict:
        """
        Detecta 1º ciclo com validação robusta
        """
        if df.empty or len(df) < 3:
            logger.error("DataFrame insuficiente para detecção de ciclo")
            return self._build_empty_signal()
        
        current_high = df['high'].tail(3).max()
        current_low = df['low'].tail(3).min()
        current_close = df['close'].iloc[-1]
        
        # Rompimento mínimo (30% do range ou mínimo absoluto)
        rompimento_min = max(canal['range_pts'] * self.rompimento_min_pct, 10.0)
        
        long_rompeu = (current_high > canal['high'] + rompimento_min)
        short_rompeu = (current_low < canal['low'] - rompimento_min)
        
        # Força do rompimento (0-1 scale)
        long_strength = (current_high - canal['high']) / canal['range_pts'] if canal['range_pts'] > 0 else 0
        short_strength = (canal['low'] - current_low) / canal['range_pts'] if canal['range_pts'] > 0 else 0
        
        # Dentro do canal (com margem de 0.2%)
        dentro_canal = (canal['low'] * 0.998 <= current_close <= canal['high'] * 1.002)
        
        sinal = {
            'current_close': current_close,
            'long_rompimento': long_rompeu,
            'short_rompimento': short_rompeu,
            'dentro_canal': dentro_canal,
            'forca_rompimento': max(long_strength, short_strength),
            'long_strength': long_strength,
            'short_strength': short_strength,
            'rompimento_min_required': rompimento_min,
            'rompimento_actual_long': current_high - canal['high'],
            'rompimento_actual_short': canal['low'] - current_low
        }
        
        logger.info(f"1º Ciclo: Long={long_rompeu} Short={short_rompeu} Força={sinal['forca_rompimento']:.2f}")
        return sinal
    
    def _build_empty_signal(self) -> Dict:
        """Sinal vazio para erros"""
        return {
            'current_close': 0,
            'long_rompimento': False,
            'short_rompimento': False,
            'dentro_canal': False,
            'forca_rompimento': 0,
            'long_strength': 0,
            'short_strength': 0,
            'rompimento_min_required': 0,
            'rompimento_actual_long': 0,
            'rompimento_actual_short': 0
        }

class GestaoFimathe:
    """
    GESTÃO SEMANAL FIMATHE: 2 stops = PARA 
    COM BLINDAGEM: Position sizing correto, ATR dinâmico, trailing stop
    """
    
    def __init__(self, capital: float = 15000, risk_pct: float = 0.01, atr_multiplier: float = 2.5):
        self.capital = capital
        self.risk_pct = risk_pct
        self.max_stops_semana = 2
        self.stops_semana = 0
        self.nivel_trader = 1  # 1=Iniciante, 2=Intermediário, 3=Avançado
        self.atr_multiplier = atr_multiplier
        self.last_reset_date = date.today()
        
        # Configurações de lotes
        self.min_lot = 0.01
        self.max_lot = 5.0
        
        logger.info(f"GestaoFimathe initialized: Capital=${capital}, Risk={risk_pct*100}%, ATRx{atr_multiplier}")
    
    def reset_semanal(self):
        """Reset semanal automático baseado em data"""
        today = date.today()
        
        # Reset toda segunda-feira
        if today.weekday() == 0 and today != self.last_reset_date:
            self.stops_semana = 0
            self.last_reset_date = today
            logger.info("Weekly stop counter reset (Monday)")
    
    def calculate_position_size(self, entry: float, stop_loss: float, symbol: str = "XAUUSD") -> float:
        """
        Calcula tamanho da posição com precisão institucional
        Considera: risco 1%, pip value real, limites de lote
        """
        self.reset_semanal()
        
        # Verificar se pode operar
        if not self.pode_operar():
            logger.warning("Cannot calculate position - weekly stop limit reached")
            return 0.0
        
        # Calcular risco em moeda
        risk_amount = self.capital * self.risk_pct
        
        # Calcular distância do stop em pontos
        risk_points = abs(entry - stop_loss)
        
        if risk_points <= 0:
            logger.error(f"Invalid risk points: {risk_points}")
            return self.min_lot
        
        try:
            # Obter informações reais do símbolo
            symbol_info = mt5.symbol_info(symbol)
            if symbol_info is None:
                logger.error(f"Symbol info not available for {symbol}")
                return self.min_lot
            
            # Calcular valor por ponto
            point = symbol_info.point
            tick_value = symbol_info.trade_tick_value
            tick_size = symbol_info.trade_tick_size
            
            if point <= 0 or tick_value <= 0:
                logger.warning(f"Invalid symbol metrics: point={point}, tick_value={tick_value}")
                # Fallback para XAUUSD padrão
                point = 0.01
                tick_value = 1.0
            
            # Cálculo preciso do lote
            # risk_amount = lots * risk_points * tick_value / point
            # lots = risk_amount * point / (risk_points * tick_value)
            
            lots = (risk_amount * point) / (risk_points * tick_value)
            
            # Aplicar limites
            lots = max(self.min_lot, min(lots, self.max_lot))
            lots = round(lots, 2)  # Arredondar para 2 casas decimais
            
            logger.info(f"Position size: {lots:.2f} lots (Risk: ${risk_amount:.2f}, Points: {risk_points:.1f})")
            return lots
            
        except Exception as e:
            logger.error(f"Position size calculation error: {e}")
            # Fallback conservador
            return self.min_lot
    
    def calculate_dynamic_stop_loss(self, entry: float, atr_value: float, direcao: str) -> float:
        """
        Stop Loss dinâmico baseado em ATR
        ATR multiplier ajustável conforme volatilidade
        """
        if atr_value <= 0:
            logger.warning(f"Invalid ATR value: {atr_value}, using minimum")
            atr_value = 10.0  # Mínimo seguro
        
        stop_offset = atr_value * self.atr_multiplier
        
        # Garantir offset mínimo
        min_offset = 50.0  # 50 pontos mínimo
        stop_offset = max(stop_offset, min_offset)
        
        if direcao == 'LONG':
            stop_loss = entry - stop_offset
            logger.info(f"Dynamic SL LONG: Entry ${entry:.2f} - {stop_offset:.1f}pts = ${stop_loss:.2f}")
        else:  # SHORT
            stop_loss = entry + stop_offset
            logger.info(f"Dynamic SL SHORT: Entry ${entry:.2f} + {stop_offset:.1f}pts = ${stop_loss:.2f}")
        
        return stop_loss
    
    def calculate_trailing_stop(self, current_price: float, entry: float, current_stop: float, 
                               direcao: str, atr_value: float) -> float:
        """
        Trailing Stop baseado em ATR
        Atualiza stop conforme preço se move a favor
        """
        if atr_value <= 0:
            return current_stop
        
        trail_offset = atr_value * self.atr_multiplier * 0.8  # 80% do SL inicial
        
        if direcao == 'LONG':
            new_stop = current_price - trail_offset
            # Só move para cima
            if new_stop > current_stop:
                logger.info(f"Trailing SL updated: ${current_stop:.2f} → ${new_stop:.2f}")
                return new_stop
        else:  # SHORT
            new_stop = current_price + trail_offset
            # Só move para baixo
            if new_stop < current_stop:
                logger.info(f"Trailing SL updated: ${current_stop:.2f} → ${new_stop:.2f}")
                return new_stop
        
        return current_stop
    
    def take_profit_levels(self, canal: Dict, direcao: str) -> Tuple[float, Optional[float], Optional[float]]:
        """
        Take Profit em 3 níveis conforme nível do trader
        Nível 1: R:R 1:1 (iniciante)
        Nível 2: R:R 1:1 + 1:1.5 (intermediário)
        Nível 3: R:R 1:1 + 1:1.5 + 1:2 (avançado)
        """
        range_canal = canal['range_pts']
        
        if direcao == 'LONG':
            take1 = canal['high'] + range_canal  # R:R 1:1
            
            if self.nivel_trader >= 2:
                take2 = canal['high'] + range_canal * 1.5  # R:R 1:1.5
            else:
                take2 = None
            
            if self.nivel_trader >= 3:
                take3 = canal['high'] + range_canal * 2.0  # R:R 1:2
            else:
                take3 = None
                
        else:  # SHORT
            take1 = canal['low'] - range_canal  # R:R 1:1
            
            if self.nivel_trader >= 2:
                take2 = canal['low'] - range_canal * 1.5  # R:R 1:1.5
            else:
                take2 = None
            
            if self.nivel_trader >= 3:
                take3 = canal['low'] - range_canal * 2.0  # R:R 1:2
            else:
                take3 = None
        
        logger.info(f"Take Profit levels: TP1={take1:.2f}, TP2={take2 if take2 else 'N/A'}, TP3={take3 if take3 else 'N/A'}")
        return take1, take2, take3
    
    def pode_operar(self) -> bool:
        """2 stops/semana = PARA (regra Fimathe)"""
        return self.stops_semana < self.max_stops_semana
    
    def register_stop(self):
        """Registra stop hit (para contagem semanal)"""
        self.stops_semana += 1
        logger.warning(f"Stop hit registered: {self.stops_semana}/{self.max_stops_semana} this week")
        
        if self.stops_semana >= self.max_stops_semana:
            logger.critical("WEEKLY STOP LIMIT REACHED - Trading halted until Monday")

    def execute_trading_cycle(self, df: pd.DataFrame, is_domingo_m15: bool = False) -> Dict:
        """
        Executa ciclo completo de trading com blindagem
        Canal → Ciclo → Gestão → Execução
        """
        # Verificar inicialização
        if not self.system_initialized:
            logger.error("System not initialized. Call initialize_system() first.")
            return {"status": "SYSTEM_NOT_INITIALIZED"}
        
        # Verificar blindagem (se ativa)
        if self.enable_shield and not self.shield_active:
            logger.critical("Shield not active - trading cycle aborted")
            return {"status": "SHIELD_NOT_ACTIVE"}
        
        # Verificar gestão semanal
        if not self.gestao.pode_operar():
            logger.warning("Weekly stop limit reached - trading halted")
            return {"status": "MAX_STOPS_SEMANA", "stops": self.gestao.stops_semana}
        
        self.operation_count += 1
        
        # 0. CÁLCULO ATR (para SL Dinâmico)
        atr_value = calculate_atr(df, period=14)
        if atr_value <= 0:
            logger.error("ATR calculation failed")
            return {"status": "ATR_ERROR", "operation": self.operation_count}
        
        # 1. CANAL ABERTURA (4 velas)
        canal = self.canal.build(df, is_domingo_m15)
        if not canal['valid']:
            return {"status": "CANAL_INVALIDO", "canal": canal, "operation": self.operation_count}
        
        # 2. 1º CICLO
        ciclo = self.ciclo.detect(canal, df)
        
        sinal_status = "AGUARDANDO_CICLO"
        resultado = {
            "operation_id": f"OP-{self.operation_count:04d}",
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "canal": {k: v for k, v in canal.items() if k != 'validation_errors'},
            "ciclo": ciclo,
            "atr_value": atr_value,
            "status": sinal_status,
            "dry_run": self.dry_run,
            "shield_active": self.shield_active,
            "verification_id": self.verification_id
        }
        
        # 3. EXECUÇÃO LONG
        if ciclo['long_rompimento']:
            return self._execute_long_trade(canal, atr_value, is_domingo_m15, resultado)
        
        # 4. EXECUÇÃO SHORT
        elif ciclo['short_rompimento']:
            return self._execute_short_trade(canal, atr_value, is_domingo_m15, resultado)
        
        # 5. SEM SINAL
        logger.info(f"Operation {self.operation_count}: No valid signal")
        return resultado
    
    def _execute_long_trade(self, canal: Dict, atr_value: float, 
                           is_domingo_m15: bool, base_result: Dict) -> Dict:
        """Executa trade LONG com blindagem completa"""
        entry = canal['high']
        
        # SL Dinâmico
        stop_loss = self.gestao.calculate_dynamic_stop_loss(entry, atr_value, 'LONG')
        risk_pips = abs(entry - stop_loss)
        
        # Take Profit levels
        take1, take2, take3 = self.gestao.take_profit_levels(canal, 'LONG')
        
        # Position sizing
        lots = self.gestao.calculate_position_size(entry, stop_loss, self.symbol)
        
        # Risk/Reward
        rr1 = (take1 - entry) / risk_pips if risk_pips > 0 else 0
        
        # Montar resultado
        resultado = {
            **base_result,
            'status': "LONG_1CICLO",
            'direction': 'LONG',
            'entry': entry,
            'stop_loss': stop_loss,
            'take_profit_1': take1,
            'take_profit_2': take2,
            'take_profit_3': take3,
            'lots': lots,
            'risk_pips': risk_pips,
            'risk_reward_1': rr1,
            'potential_profit_1': (take1 - entry) * lots * 100 if lots > 0 else 0,
            'potential_loss': risk_pips * lots * 100 if lots > 0 else 0,
            'confluencia': f"DomingoM15:{is_domingo_m15} Gap:{canal['is_gap']} ATR:{atr_value:.1f}",
            'risk_management': {
                'max_risk_pct': self.gestao.risk_pct * 100,
                'weekly_stops': f"{self.gestao.stops_semana}/{self.gestao.max_stops_semana}",
                'atr_multiplier': self.gestao.atr_multiplier
            }
        }
        
        # Executar trade (se não for dry_run)
        if not self.dry_run:
            trade_result = self._execute_mt5_trade(resultado)
            resultado['trade_execution'] = trade_result
        else:
            resultado['trade_execution'] = {"status": "DRY_RUN", "message": "Trade simulated"}
        
        logger.info(f"LONG Signal: Entry=${entry:.2f}, SL=${stop_loss:.2f}, TP1=${take1:.2f}, Lots={lots:.2f}, R:R={rr1:.2f}")
        
        # Atribuir tarefa de monitoramento
        self.multi_agent.assign_task('debugger', {
            'description': f"Monitor LONG trade {resultado['operation_id']}",
            'trade_details': resultado,
            'priority': 'HIGH'
        })
        
        return resultado
    
    def _execute_short_trade(self, canal: Dict, atr_value: float, 
                            is_domingo_m15: bool, base_result: Dict) -> Dict:
        """Executa trade SHORT com blindagem completa"""
        entry = canal['low']
        
        # SL Dinâmico
        stop_loss = self.gestao.calculate_dynamic_stop_loss(entry, atr_value, 'SHORT')
        risk_pips = abs(entry - stop_loss)
        
        # Take Profit levels
        take1, take2, take3 = self.gestao.take_profit_levels(canal, 'SHORT')
        
        # Position sizing
        lots = self.gestao.calculate_position_size(entry, stop_loss, self.symbol)
        
        # Risk/Reward
        rr1 = (entry - take1) / risk_pips if risk_pips > 0 else 0
        
        # Montar resultado
        resultado = {
            **base_result,
            'status': "SHORT_1CICLO",
            'direction': 'SHORT',
            'entry': entry,
            'stop_loss': stop_loss,
            'take_profit_1': take1,
            'take_profit_2': take2,
            'take_profit_3': take3,
            'lots': lots,
            'risk_pips': risk_pips,
            'risk_reward_1': rr1,
            'potential_profit_1': (entry - take1) * lots * 100 if lots > 0 else 0,
            'potential_loss': risk_pips * lots * 100 if lots > 0 else 0,
            'confluencia': f"DomingoM15:{is_domingo_m15} Gap:{canal['is_gap']} ATR:{atr_value:.1f}",
            'risk_management': {
                'max_risk_pct': self.gestao.risk_pct * 100,
                'weekly_stops': f"{self.gestao.stops_semana}/{self.gestao.max_stops_semana}",
                'atr_multiplier': self.gestao.atr_multiplier
            }
        }
        
        # Executar trade (se não for dry_run)
        if not self.dry_run:
            trade_result = self._execute_mt5_trade(resultado)
            resultado['trade_execution'] = trade_result
        else:
            resultado['trade_execution'] = {"status": "DRY_RUN", "message": "Trade simulated"}
        
        logger.info(f"SHORT Signal: Entry=${entry:.2f}, SL=${stop_loss:.2f}, TP1=${take1:.2f}, Lots={lots:.2f}, R:R={rr1:.2f}")
        
        # Atribuir tarefa de monitoramento
        self.multi_agent.assign_task('debugger', {
            'description': f"Monitor SHORT trade {resultado['operation_id']}",
            'trade_details': resultado,
            'priority': 'HIGH'
        })
        
        return resultado
    
    def _execute_mt5_trade(self, trade_data: Dict) -> Dict:
        """
        Executa trade real no MT5
        Implementação básica - expandir conforme necessidade
        """
        try:
            # Preparar requisição de trade
            symbol = self.symbol
            order_type = mt5.ORDER_TYPE_BUY if trade_data['direction'] == 'LONG' else mt5.ORDER_TYPE_SELL
            volume = trade_data['lots']
            price = mt5.symbol_info_tick(symbol).ask if trade_data['direction'] == 'LONG' else mt5.symbol_info_tick(symbol).bid
            
            # Configurar SL e TP
            sl = trade_data['stop_loss']
            tp = trade_data['take_profit_1']
            
            # Criar request
            request = {
                "action": mt5.TRADE_ACTION_DEAL,
                "symbol": symbol,
                "volume": volume,
                "type": order_type,
                "price": price,
                "sl": sl,
                "tp": tp,
                "deviation": 10,
                "magic": 234000,
                "comment": f"FimatheTIER0 {trade_data['operation_id']}",
                "type_time": mt5.ORDER_TIME_GTC,
                "type_filling": mt5.ORDER_FILLING_IOC,
            }
            
            # Enviar ordem
            result = mt5.order_send(request)
            
            trade_result = {
                "retcode": result.retcode,
                "order_id": result.order if result else None,
                "volume_executed": result.volume if result else 0,
                "price_executed": result.price if result else 0,
                "comment": result.comment if result else "",
                "success": result.retcode == mt5.TRADE_RETCODE_DONE if result else False
            }
            
            if trade_result["success"]:
                logger.info(f"Trade executed successfully: Order #{trade_result['order_id']}")
            else:
                logger.error(f"Trade execution failed: {trade_result}")
                
                # Registrar como stop (se falha)
                self.gestao.register_stop()
            
            return trade_result
            
        except Exception as e:
            logger.error(f"Trade execution error: {e}")
            return {"error": str(e), "success": False}
    
    def emergency_shutdown(self, reason: str = "SYSTEM_FAILURE"):
        """
        Shutdown de emergência do sistema
        Conforme Emergency Protocol do Omega
        """
        logger.critical(f"EMERGENCY SHUTDOWN INITIATED: {reason}")
        
        # 1. Fechar todas as posições abertas
        if not self.dry_run and mt5.initialize():
            positions = mt5.positions_get(symbol=self.symbol)
            if positions:
                for position in positions:
                    # Fechar posição
                    close_request = {
                        "action": mt5.TRADE_ACTION_DEAL,
                        "symbol": position.symbol,
                        "volume": position.volume,
                        "type": mt5.ORDER_TYPE_BUY if position.type == mt5.POSITION_TYPE_SELL else mt5.ORDER_TYPE_SELL,
                        "position": position.ticket,
                        "price": mt5.symbol_info_tick(position.symbol).bid if position.type == mt5.POSITION_TYPE_SELL else mt5.symbol_info_tick(position.symbol).ask,
                        "deviation": 20,
                        "magic": 234000,
                        "comment": f"EmergencyClose_{reason}",
                        "type_time": mt5.ORDER_TIME_GTC,
                        "type_filling": mt5.ORDER_FILLING_IOC,
                    }
                    
                    mt5.order_send(close_request)
                    logger.warning(f"Emergency closed position #{position.ticket}")
            
            mt5.shutdown()
        
        # 2. Gerar relatório de emergência
        emergency_report = {
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "system": "FIMATHE_CORE_TIER0_v2.0",
            "shutdown_reason": reason,
            "operation_count": self.operation_count,
            "shield_active": self.shield_active,
            "verification_id": self.verification_id,
            "weekly_stops": f"{self.gestao.stops_semana}/{self.gestao.max_stops_semana}",
            "capital": self.capital,
            "recommendations": [
                "Review system logs",
                "Check market conditions",
                "Verify system integrity",
                "Contact administrator if needed"
            ]
        }
        
        with open("emergency_shutdown_report.json", 'w') as f:
            json.dump(emergency_report, f, indent=2)
        
        # 3. Notificar agentes
        for agent in self.multi_agent.AGENTS:
            self.multi_agent.assign_task(agent, {
                'description': f"Emergency shutdown: {reason}",
                'priority': 'CRITICAL',
                'action_required': 'REVIEW_AND_REPORT'
            })
        
        logger.critical("EMERGENCY SHUTDOWN COMPLETE - System halted")
