### **1. FIN-SENSE Pipeline**

**Objetivo:** Ingestão de dados do `demo_log_swing_trade` e cálculo de métricas institucionais.  
**Código:**

```python
# modules/DOS_MODULE/pipeline.py
from typing import Dict, Union, Optional
import pandas as pd
import numpy as np
from datetime import datetime
import hashlib
from .metrics.institutional import Tier0Metrics
from .config import REQUIRED_SWING_COLS

def run_dos_pipeline(
    market: Optional[Union[pd.DataFrame, str]] = None,
    source: str = "frame",
    notional_usd: float = 1_000_000.0,
    limit_rows: Optional[int] = None
) -> Dict:
    """
    Pipeline FIN-SENSE: demo_log_swing_trade → métricas Tier-0.
    Args:
        market: DataFrame ou caminho para CSV.
        source: "frame" ou "postgres".
        notional_usd: Capital base para métricas.
        limit_rows: Limite de linhas (para testes).
    Returns:
        Dict com VaR, CVaR, RAROC, etc.
    """
    if source == "frame":
        df = market.copy() if isinstance(market, pd.DataFrame) else pd.read_csv(market)
    else:
        # Lógica para PostgreSQL (FIN-SENSE)
        from .bridge_fin_sense import get_fin_sense_data
        df = get_fin_sense_data(limit=limit_rows)

    # Validação de colunas
    missing = [col for col in REQUIRED_SWING_COLS if col not in df.columns]
    if missing:
        raise ValueError(f"Colunas obrigatórias ausentes: {missing}")

    # Cálculo de retornos sintéticos (até integração com livro real)
    df['returns'] = df['spread'].pct_change().fillna(0)

    # Métricas Tier-0
    metrics = Tier0Metrics(
        returns=df['returns'],
        exposure=notional_usd,
        confidence=0.95
    )
    result = metrics.full_analysis()

    # Proveniência e integridade
    result.update({
        "input_sha256": hashlib.sha256(pd.util.hash_pandas_object(df).values).hexdigest(),
        "timestamp": datetime.now().isoformat(),
        "source": source,
        "rows_processed": len(df)
    })

    return result
```

**Exemplo de Uso:**

```python
from omega_dos.pipeline import run_dos_pipeline

# De um DataFrame

# De um CSV
```

### **2. DOS-TRADING V1.0 (MT5)**

**Objetivo:** 4 camadas de interpretação sobre dados MT5 (OHLC).  
**Código:**

```python
# modules/DOS_MODULE/trading/dos_trading_v1.py
import pandas as pd
import numpy as np
import os
from pathlib import Path
from typing import Dict, List, Tuple
from ..config import DOS_MT5_DIR
from ..metrics.risk import calculate_var

class DOS_TRADING_V1:
    def __init__(self, mt5_dir: str = DOS_MT5_DIR):
        self.mt5_dir = Path(mt5_dir)
        if not self.mt5_dir.exists():
            raise FileNotFoundError(f"Diretório MT5 não encontrado: {mt5_dir}")

    def _load_mt5_data(self, symbol: str, year: int) -> pd.DataFrame:
        """Carrega dados MT5 (CSV: time,open,high,low,close)"""
        path = self.mt5_dir / f"{symbol}_M1_{year}.csv"
        if not path.exists():
            raise FileNotFoundError(f"Arquivo não encontrado: {path}")
        return pd.read_csv(path, parse_dates=['time'])

    def _calculate_layers(self, df: pd.DataFrame) -> pd.DataFrame:
        """Camadas 1-4: preço, microestrutura, regime, sinal composto"""
        # Camada 1: Preço puro
        df['mid'] = (df['high'] + df['low']) / 2

        # Camada 2: Microestrutura
        df['spread_bp'] = (df['high'] - df['low']) / df['mid'] * 10000
        df['tick_velocity'] = df['close'].diff().abs().rolling(10).mean()

        # Camada 3: Regime de mercado
        df['volatility'] = df['spread_bp'].rolling(100).std()
        df['trend'] = df['close'].rolling(50).mean() > df['close'].rolling(200).mean()
        df['regime'] = np.where(df['volatility'] > 3, 'high_vol',
                               np.where(df['trend'], 'uptrend', 'downtrend'))

        # Camada 4: Sinal composto
        df['signal_strength'] = (df['spread_bp'] / df['volatility']) * (1 if df['trend'] else -1)
        return df

    def run_full_pipeline(self, symbol: str, year: int = 2026) -> Dict:
        """Pipeline completo: OHLC → camadas → sinais → diagnóstico"""
        df = self._load_mt5_data(symbol, year)
        df = self._calculate_layers(df)

        # Diagnóstico de falsos negativos
        signals = df[df['signal_strength'].abs() > 2.0]  # Threshold
        false_negatives = len(df[df['regime'] == 'high_vol']) - len(signals)

        # Backtest simplificado
        df['position'] = np.where(df['signal_strength'] > 2, 1,
                                 np.where(df['signal_strength'] < -2, -1, 0))
        df['returns'] = df['close'].pct_change().shift(-1) * df['position']
        pnl = df['returns'].cumsum()

        return {
            "symbol": symbol,
            "year": year,
            "signals_generated": len(signals),
            "false_negatives": false_negatives,
            "pnl_curve": pnl.tolist(),
            "max_drawdown": pnl.min(),
            "sharpe_ratio": pnl.mean() / pnl.std() * np.sqrt(252),
            "regime_distribution": df['regime'].value_counts().to_dict()
        }
```

**Exemplo de Uso:**

```python
from omega_dos.trading.dos_trading_v1 import DOS_TRADING_V1

report = trader.run_full_pipeline("XAUUSD", 2026)
```

### **3. Métricas Tier-0**

**Objetivo:** Cálculo de VaR, CVaR, RAROC, Sortino, Omega, Sharpe, Calmar.  
**Código:**

```python
# modules/DOS_MODULE/metrics/institutional.py
import numpy as np
import pandas as pd
from typing import Union, Dict
from .risk import calculate_var, calculate_cvar

class Tier0Metrics:
    def __init__(
        self,
        returns: Union[pd.Series, np.ndarray],
        exposure: float = 1_000_000.0,
        confidence: float = 0.95,
        risk_free_rate: float = 0.02
    ):
        self.returns = pd.Series(returns)
        self.exposure = exposure
        self.confidence = confidence
        self.risk_free = risk_free_rate

    def _annualize_returns(self, returns: pd.Series, periods_per_year: int = 252) -> float:
        return (1 + returns).prod() ** (periods_per_year/len(returns)) - 1

    def _annualize_volatility(self, returns: pd.Series, periods_per_year: int = 252) -> float:
        return returns.std() * np.sqrt(periods_per_year)

    def full_analysis(self) -> Dict:
        """Retorna todas as métricas Tier-0"""
        annual_return = self._annualize_returns(self.returns)
        annual_vol = self._annualize_volatility(self.returns)
        sharpe = (annual_return - self.risk_free) / annual_vol if annual_vol > 0 else 0

        var_95 = calculate_var(self.returns, self.confidence) * self.exposure
        cvar_95 = calculate_cvar(self.returns, self.confidence) * self.exposure

        # RAROC: Retorno / (Capital + Exposição × 8%)
        capital_charge = 0.08 * self.exposure
        raroc = annual_return / (1 + capital_charge) if (1 + capital_charge) > 0 else 0

        # Sortino (apenas downside risk)
        downside_returns = self.returns[self.returns < 0]
        downside_vol = self._annualize_volatility(downside_returns) if len(downside_returns) > 0 else 0
        sortino = (annual_return - self.risk_free) / downside_vol if downside_vol > 0 else 0

        # Omega Ratio (retorno positivo / negativo)
        positive = self.returns[self.returns > 0].mean() if len(self.returns[self.returns > 0]) > 0 else 0
        negative = abs(self.returns[self.returns < 0].mean()) if len(self.returns[self.returns < 0]) > 0 else 1e-6
        omega = positive / negative if negative > 0 else np.inf

        # Calmar Ratio (retorno anual / max drawdown)
        cumulative = (1 + self.returns).cumprod()
        drawdown = (cumulative.cummax() - cumulative).div(cumulative.cummax())
        max_dd = drawdown.max()
        calmar = annual_return / max_dd if max_dd > 0 else np.inf

        return {
            "annual_return": annual_return,
            "annual_volatility": annual_vol,
            "sharpe_ratio": sharpe,
            "var_95": var_95,
            "cvar_95": cvar_95,
            "raroc": raroc,
            "sortino_ratio": sortino,
            "omega_ratio": omega,
            "calmar_ratio": calmar,
            "max_drawdown": max_dd
        }
```

### **4. Backtrader (Opcional)**

**Objetivo:** Integração com ambiente de backtesting.  
**Código:**

```python
# modules/DOS_MODULE/trading/backtrader_dos.py
import backtrader as bt
import pandas as pd
from typing import Dict

def run_backtrader_mt5(csv_path: str, symbol: str = "XAUUSD") -> Dict:
    """Backtest com Backtrader (opcional)"""
    data = pd.read_csv(csv_path, parse_dates=['time'], index_col='time')
    data = data.rename(columns={
        'open': 'Open', 'high': 'High',
        'low': 'Low', 'close': 'Close', 'volume': 'Volume'
    })

    class DOSStrategy(bt.Strategy):
        def __init__(self):
            self.sma = bt.indicators.SMA(period=20)

        def next(self):
            if not self.position:
                if self.data.close[0] > self.sma[0]:
                    self.buy()
            else:
                if self.data.close[0] < self.sma[0]:
                    self.close()

    cerebro = bt.Cerebro()
    data_feed = bt.feeds.PandasData(dataname=data)
    cerebro.adddata(data_feed)
    cerebro.addstrategy(DOSStrategy)
    cerebro.run()

    return {
        "final_value": cerebro.broker.getvalue(),
        "sharpe_ratio": cerebro.strategies[0].analyzers.sharpe.get_analysis()['sharperatio'],
        "max_drawdown": cerebro.strategies[0].analyzers.drawdown.get_analysis()['max']['drawdown']
    }
```

**Exemplo de Uso:**


## **V. LIMITAÇÕES E RISCOS**


| **Tópico**                    | **Severidade** | **Descrição**                                                  | **Mitigação**                                  |
| ----------------------------- | -------------- | -------------------------------------------------------------- | ---------------------------------------------- |
| PnL sintético FIN-SENSE       | Média          | Dados sintéticos até integração com livro real.                | Validar com dados reais assim que disponíveis. |
| Backtest DOS-TRADING          | Média          | Modelo simplificado (slippage fixo, saída por janela).         | Usar apenas para diagnóstico, não produção.    |
| VaR/CVaR                      | Média          | Baseado em amostra histórica; não cobre eventos extremos.      | Complementar com stress testing.               |
| Dependência Backtrader        | Baixa          | Opcional; versão fixada em `pyproject.toml`.                   | Validar ambiente antes de CI.                  |
