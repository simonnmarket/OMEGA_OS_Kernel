def mt5_drift_test(r1y_timestamp, r1x_timestamp, tolerance_ms=1000):
    """IEEE 1012 Test Case: Timestamp sync failure."""
    drift_ms = abs(r1y_timestamp - r1x_timestamp) / 1e6  # μs → ms
    
    if drift_ms > tolerance_ms:
        return {
            'status': 'DRIFT_DETECTED',
            'drift_ms': drift_ms,
            'cointegração_quebrada': True,
            'recommended_action': 'SKIP_BAR'
        }
    return {'status': 'SYNC_OK', 'cointegração_quebrada': False}
