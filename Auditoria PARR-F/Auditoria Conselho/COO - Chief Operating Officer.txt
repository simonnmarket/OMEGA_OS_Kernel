"""
FIN-SENSE Pipeline — run_dos_pipeline()
Especificação: DOC-OFC-DOS-TRADING-V1.0-20260405-001
Entrada: bronze.demo_log_swing_trade | DataFrame custom
"""

import pandas as pd
import numpy as np
from typing import Dict, Union, Optional
from dataclasses import dataclass
import hashlib
from pathlib import Path
import logging

logger = logging.getLogger(__name__)

@dataclass
class DosPipelineResult:
    var_95_usd: float
    cvar_95_usd: float
    raroc_pct: float
    regime_label: str
    cluster_id: int
    provenance_sha256: str
    records_processed: int
    spec_id: str = __spec_id__

class FIN_SENSE_PIPELINE:
    REQUIRED_SWING_COLS = ['y', 'x', 'spread', 'z', 'beta', 'flags', 
                          'ram_mb', 'cpu_pct', 'proc_ms', 'opp_cost']

    @classmethod
    def sha256_canonical(cls, df: pd.DataFrame) -> str:
        """Hash determinístico OHLC/inputs (Tier-0 audit)"""
        canonical = df[cls.REQUIRED_SWING_COLS].round(6).to_json(orient='records', date_format='iso')
        return hashlib.sha256(canonical.encode()).hexdigest()[:64]

    @classmethod
    def synthetic_positions_trades_from_swing(cls, swing_df: pd.DataFrame, 
                                            notional_usd: float = 10000.0) -> pd.DataFrame:
        """PnL sintético FIN-SENSE → posições/trades"""
        positions = swing_df.copy()
        positions['market_value'] = notional_usd
        positions['cost_basis'] = notional_usd * (1 - positions['spread']/2)
        positions['pnl_unrealized'] = positions['y'] * notional_usd / 100  # 'y' como % PnL
        return positions

    @classmethod
    def calculate_var_cvar(cls, positions: pd.DataFrame, confidence: float = 0.95) -> tuple:
        """VaR/CVaR histórico sobre PnL sintético"""
        returns = positions['pnl_unrealized'].pct_change().dropna()
        if len(returns) < 10:
            return 0.0, 0.0
        
        var_pct = -np.percentile(returns, (1-confidence)*100)
        var_usd = abs(var_pct * positions['market_value'].iloc[-1])
        
        tail_losses = returns[returns <= -var_pct]
        cvar_usd = abs(-tail_losses.mean() * positions['market_value'].iloc[-1]) if len(tail_losses) > 0 else var_usd * 1.2
        
        return var_usd, cvar_usd

    @classmethod
    def run_dos_pipeline(cls, market: Union[str, pd.DataFrame], 
                        source: str = "frame", notional_usd: float = 10000.0,
                        limit_rows: Optional[int] = None) -> DosPipelineResult:
        """
        FIN-SENSE Pipeline principal — Tier-0 determinístico
        Args:
            market: DataFrame swing ou "postgres"
            source: "frame" | "postgres"
            notional_usd: Exposição para métricas
        """
        logger.info(f"🚀 FIN-SENSE Pipeline | source={source} | notional=${notional_usd:,.0f}")
        
        if source == "postgres":
            # Placeholder Postgres (implementar get_connection_dsn())
            logger.warning("⚠️ Postgres requer FIN_SENSE_DSN")
            market = pd.DataFrame()  # Stub
        
        elif source == "frame":
            if not isinstance(market, pd.DataFrame):
                raise ValueError("market deve ser DataFrame em source='frame'")
            
            missing_cols = [col for col in cls.REQUIRED_SWING_COLS if col not in market.columns]
            if missing_cols:
                raise ValueError(f"Faltam colunas swing: {missing_cols}")
            
            if limit_rows:
                market = market.head(limit_rows)
        
        else:
            raise ValueError(f"Source inválido: {source}")
        
        if market.empty:
            raise ValueError("Market DataFrame vazio")
        
        # Proveniência SHA-256 (Tier-0 auditabilidade)
        provenance = cls.sha256_canonical(market)
        
        # Posições sintéticas
        positions = cls.synthetic_positions_trades_from_swing(market, notional_usd)
        
        # Métricas institucionais
        var_95, cvar_95 = cls.calculate_var_cvar(positions)
        exposure = positions['market_value'].sum()
        pnl_estimate = positions['pnl_unrealized'].sum()
        risk_capital = 1000000 + exposure * 0.08
        raroc = (pnl_estimate / risk_capital * 100) if risk_capital > 0 else 0
        
        # Regime/cluster simplificado
        vol_regime = 'medium'  # Stub até microestrutura real
        cluster_id = 1
        
        logger.info(f"✅ FIN-SENSE completo | VaR=${var_95:.0f} | RAROC={raroc:.1f}%")
        
        return DosPipelineResult(
            var_95_usd=var_95,
            cvar_95_usd=cvar_95,
            raroc_pct=raroc,
            regime_label=vol_regime,
            cluster_id=cluster_id,
            provenance_sha256=provenance,
            records_processed=len(market)
        )

def run_dos_pipeline(*args, **kwargs):
    """Facade principal"""
    return FIN_SENSE_PIPELINE.run_dos_pipeline(*args, **kwargs)

        risk_capital = self.exposure * 0.08  # Basel III
        return (excess_ret * 252 / risk_capital) * 100

        var_threshold = -np.percentile(self.returns, (1-confidence)*100)
        tail = self.returns[self.returns <= -var_threshold]
        return -tail.mean() if len(tail) > 0 else 0

#!/usr/bin/env python3
"""
ETAPA 2: DOS-TRADING V1.0 — 4 Camadas MT5 OHLC
Camada1: Preço | Camada2: Microestrutura | Camada3: Regime | Camada4: Sinais
"""
import pandas as pd
import numpy as np
from pathlib import Path
from dataclasses import dataclass
from typing import List, Dict
import logging

logger = logging.getLogger(__name__)

@dataclass
class TradingSignal:
    timestamp: str
    symbol: str
    regime: str
    signal_strength: float
    entry_price: float
    direction: int

class DOS_TRADING_V1:
    MT5_COLS = ['time', 'open', 'high', 'low', 'close']
    
    def __init__(self, mt5_dir: str = "mt5_data"):
        self.mt5_dir = Path(mt5_dir)
    
    def load_mt5(self, symbol: str, years: List[int] = [2024,2025,2026]) -> pd.DataFrame:
        dfs = []
        for year in years:
            csv = self.mt5_dir / f"{symbol}_M1_{year}.csv"
            if csv.exists():
                df = pd.read_csv(csv, parse_dates=['time'])
                dfs.append(df)
        return pd.concat(dfs, ignore_index=True) if dfs else pd.DataFrame()
    
    def layer1_price(self, data: pd.DataFrame) -> pd.DataFrame:
        df = data.copy()
        df['body_size'] = abs(df['close'] - df['open'])
        df['range_size'] = df['high'] - df['low']
        df['price_ma5'] = df['close'].rolling(5).mean()
        df['price_ma20'] = df['close'].rolling(20).mean()
        df['price_momentum'] = (df['close'] - df['price_ma20']) / df['price_ma20']
        return df
    
    def layer2_micro(self, data: pd.DataFrame) -> pd.DataFrame:
        df = data.copy()
        df['spread_bp'] = ((df['high'] - df['low']) / df['close']) * 10000
        df['volume_proxy'] = df['range_size'] * df['body_size']
        df['spread_ma'] = df['spread_bp'].rolling(20).mean()
        return df
    
    def layer3_regime(self, data: pd.DataFrame) -> pd.DataFrame:
        df = data.copy()
        df['volatility'] = df['close'].pct_change().rolling(20).std()
        df['vol_regime'] = pd.cut(df['volatility'], 
                                [0, 0.005, 0.015, np.inf], 
                                labels=['low', 'medium', 'high'])
        df['price_trend'] = np.sign(df['price_ma5'] - df['price_ma20'])
        return df
    
    def layer4_signals(self, data: pd.DataFrame) -> pd.DataFrame:
        df = data.copy()
        weights = {'price': 0.3, 'micro': 0.25, 'regime': 0.45}
        
        df['signal_price'] = np.where(df['price_momentum'] > 0.002, 1,
                                    np.where(df['price_momentum'] < -0.002, -1, 0))
        df['signal_micro'] = np.where((df['spread_ma'] < df['spread_ma'].quantile(0.3)) &
                                    (df['volume_proxy'] > df['volume_proxy'].rolling(20).mean()), 1, 0)
        df['signal_regime'] = np.where(df['vol_regime'] == 'low', df['price_trend'], 0)
        
        df['composite_signal'] = (df['signal_price'] * weights['price'] + 
                                df['signal_micro'] * weights['micro'] + 
                                df['signal_regime'] * weights['regime'])
        df['signal_strength'] = np.abs(df['composite_signal'])
        df['direction'] = np.sign(df['composite_signal'])
        return df
    
    def run_full_pipeline(self, symbol: str = "XAUUSD") -> Dict:
        data = self.load_mt5(symbol)
        if data.empty:
            return {'status': 'NO_DATA'}
        
        # 4 Camadas sequenciais
        data = self.layer1_price(data)
        data = self.layer2_micro(data)
        data = self.layer3_regime(data)
        data = self.layer4_signals(data)
        
        strong_signals = data[data['signal_strength'] > 0.6]
        signals = [TradingSignal(
            t.isoformat(), symbol, r, s, c, d
        ) for t, r, s, c, d in zip(strong_signals['time'], 
                                 strong_signals['vol_regime'], 
                                 strong_signals['signal_strength'],
                                 strong_signals['close'],
                                 strong_signals['direction'])]
        
        return {
            'status': 'SUCCESS',
            'symbol': symbol,
            'signals_generated': len(signals),
            'strongest_signal': max(signals, key=lambda x: x.signal_strength) if signals else None
        }

if __name__ == "__main__":
    trader = DOS_TRADING_V1()
    print(trader.run_full_pipeline())

"""

from typing import Dict

class Tier0Metrics:
    def __init__(self, returns: pd.Series, exposure_usd: float = 1_000_000):
        self.returns = returns.dropna()
        self.exposure = exposure_usd
        self.risk_free_daily = 0.04 / 252  # 4% anual
    
    def var_historical(self, confidence: float = 0.95) -> Dict:
        var_pct = -np.percentile(self.returns, (1-confidence)*100)
        return {'var_pct': var_pct, 'var_usd': abs(var_pct * self.exposure)}
    
    def cvar_historical(self, confidence: float = 0.95) -> float:
        var_threshold = -np.percentile(self.returns, (1-confidence)*100)

    def sortino(self) -> float:
        downside = self.returns[self.returns < 0]
        downside_std = downside.std() * np.sqrt(252)
        return (self.returns.mean() * 252) / downside_std if downside_std > 0 else 0
    
    def full_analysis(self) -> Dict:
        return {
            'var_95': self.var_historical(0.95),
            'var_99': self.var_historical(0.99),
            'cvar_95': self.cvar_historical(0.95),
            'raroc_pct': self.raroc(),
            'sortino': self.sortino(),
            'sharpe': (self.returns.mean() - self.risk_free_daily) / self.returns.std() * np.sqrt(252),
            'total_returns': len(self.returns)
        }

import numpy as np

class DOSStrategy(bt.Strategy):
    params = (('size', 100000), ('min_strength', 0.6))
    
    def __init__(self):
        self.layer1 = bt.indicators.SMA(self.data.close, period=5)
        self.layer2 = bt.indicators.RSI(self.data.close, period=14)
        self.vol = bt.indicators.StdDev(self.data.close, period=20)
    
    def next(self):
        price_signal = 1 if self.data.close[0] > self.layer1[0] else -1
        rsi_signal = 1 if self.layer2[0] < 30 else -1 if self.layer2[0] > 70 else 0
        vol_regime = 'high' if self.vol[0] > 0.015 else 'low'
        
        strength = abs(price_signal * 0.4 + rsi_signal * 0.3 + (1 if vol_regime == 'low' else 0) * 0.3)
        
        if strength >= self.params.min_strength and not self.position:
            size = self.params.size / self.data.close[0]
            self.buy(size=size) if price_signal > 0 else self.sell(size=size)

class DOSAnalyzer(bt.Analyzer):
    def get_analysis(self):
        pnls = np.array([trade.pnlcomm for trade in self.strategy.trades 
                        if trade.isclosed])
        if len(pnls) == 0:
            return {'trades': 0}
        
        metrics = Tier0Metrics(pd.Series(pnls))
        return {
            'total_trades': len(pnls),
            **metrics.full_analysis()
        }

def run_backtrader_demo():
    cerebro = bt.Cerebro()
    data = bt.feeds.PandasData(dataname=pd.read_csv("mt5_data/demo.csv"))
    cerebro.adddata(data)
    cerebro.addstrategy(DOSStrategy)
    cerebro.addanalyzer(DOSAnalyzer, _name="dos")
    cerebro.broker.setcash(1000000)
    result = cerebro.run()
    print(result[0].analyzers.dos.get_analysis())

if __name__ == "__main__":
    run_backtrader_demo()
