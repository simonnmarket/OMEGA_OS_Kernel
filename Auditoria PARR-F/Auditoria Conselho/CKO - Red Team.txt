import hashlib

def verify_file_integrity(filepath, expected_hash):
    sha3 = hashlib.sha3_256()
    with open(filepath, "rb") as f:
        while chunk := f.read(8192):
            sha3.update(chunk)
    calculated_hash = sha3.hexdigest()
    
    assert calculated_hash == expected_hash, f"BREACH: Hash Mismatch! \nEsp: {expected_hash}\nCalc: {calculated_hash}"
    return True

# Execução

from scipy.stats import norm

df = pd.read_csv("STRESS_V10_5_SWING_TRADE.csv")
z_abs = df['z'].abs()

# Recalcular P95 Real
calculated_p95 = z_abs.quantile(0.95)
print(f"P95 Calculado: {calculated_p95:.4f}")

# Validar a afirmativa do Dossiê
threshold = 2.0
p_value_threshold = 1 - norm.cdf(threshold)
p_value_p95 = 1 - norm.cdf(calculated_p95)

print(f"Probabilidade Gatilho (Z>=2.0): {p_value_threshold:.4%}")
print(f"Probabilidade P95 (Z>={calculated_p95}): {p_value_p95:.4%}")
