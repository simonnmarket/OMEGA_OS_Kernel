import subprocess
import sys

# O SHA aprovado pelo Conselho (f391196c8...)
APPROVED_SHA = "f391196c8"

def check_commit_sha():
    try:
        # Obtém o SHA atual do HEAD (primeiros 9 caracteres)
        result = subprocess.check_output(['git', 'rev-parse', 'HEAD'], cwd='.')
        current_sha = result.decode('utf-8').strip()[:9]
        
        if current_sha == APPROVED_SHA:
            print(f"[SECURITY CHECK] OK. Commit Ativo: {current_sha}")
            return True
        else:
            print(f"[SECURITY ALERT] VIOLAÇÃO DE PROTOCOLO.")
            print(f"Esperado: {APPROVED_SHA}...")
            print(f"Encontrado: {current_sha}...")
            print("ERRO CRÍTICO: Você está tentando rodar uma versão não aprovada.")
            return False
    except Exception as e:
        print(f"[FATAL] Erro ao verificar Git: {e}")
        return False

if __name__ == "__main__":
    if not check_commit_sha():
        sys.exit(1) # Bloqueia a execução
