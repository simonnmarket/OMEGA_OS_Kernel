//+------------------------------------------------------------------+
//| Função para verificar padrões de absorção                          |
//+------------------------------------------------------------------+
bool CheckAbsorptionPattern()
{
    double closeCurrent = Close[0];
    double closePrevious = Close[1];
    double volumeCurrent = Volume[0];
    double volumePrevious = Volume[1];

    // Condição de absorção: preço sobe com volume decrescente
    if (closeCurrent > closePrevious && volumeCurrent < volumePrevious)
    {
        // Verifica se a diferença de preço é significativa
        double priceChange = closeCurrent - closePrevious;
        if (priceChange > 0.0005) // Ajuste o valor conforme necessário
        {
            return true; // Padrão de absorção identificado
        }
    }
    return false; // Nenhum padrão encontrado
}

//+------------------------------------------------------------------+
//| Função para executar uma ordem de compra                          |
//+------------------------------------------------------------------+
void ExecuteBuyOrder()
{
    double lotSize = 1.0; // Tamanho da posição
    double stopLoss = Bid - 50 * Point; // Stop loss de 50 pips
    double takeProfit = Bid + 100 * Point; // Take profit de 100 pips

    // Envia a ordem de compra
    if (OrderSend(Symbol(), OP_BUY, lotSize, Ask, 2, stopLoss, takeProfit, "Compra", 0, 0, clrGreen) < 0)
    {
        Print("Erro ao enviar ordem: ", GetLastError());
    }
}
